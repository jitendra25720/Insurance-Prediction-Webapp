Encoding of variables¶

In [7]:
def sex_encoder(k):
    if k=='male':
        return 1
    else:
        return 0
    
def smoke_encoder(k):
    if k=='yes':
        return 1
    else:
        return 0
    
def region_encoder(k):
    if k=='southeast':
        return 0
    elif k=='southwest':
        return 1
    elif k=='northeast':
        return 2
    else:
        return 3
    
sex_encoded = df['sex'].map(sex_encoder)
smoker_encoded = df['smoker'].map(smoke_encoder)
region_encoded = df['region'].map(region_encoder)

    l1.append({'R2 score':r2_score(test_y, model), 'MAE':mean_absolute_error(test_y, model),
                            'MAPE':mean_absolute_percentage_error(test_y, model), 
                            'MSE':mean_squared_error(test_y, model),
                            'RMSE':np.sqrt(mean_squared_error(test_y, model))})
